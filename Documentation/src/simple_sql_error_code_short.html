				Usage:
					if db.last_error_code = {SIMPLE_SQL_ERROR_CODE}.constraint then
						-- handle constraint violation

feature -- Primary Result Codes

	Ok: INTEGER_32 = 0
			-- Successful result

	Error: INTEGER_32 = 1
			-- Generic error

	Internal: INTEGER_32 = 2
			-- Internal logic error in SQLite

	Perm: INTEGER_32 = 3
			-- Access permission denied

	Abort: INTEGER_32 = 4
			-- Callback routine requested an abort

	Busy: INTEGER_32 = 5
			-- The database file is locked

	Locked: INTEGER_32 = 6
			-- A table in the database is locked

	Nomem: INTEGER_32 = 7
			-- A malloc() failed

	Readonly: INTEGER_32 = 8
			-- Attempt to write a readonly database

	Interrupt: INTEGER_32 = 9
			-- Operation terminated by sqlite3_interrupt()

	Ioerr: INTEGER_32 = 10
			-- Some kind of disk I/O error occurred

	Corrupt: INTEGER_32 = 11
			-- The database disk image is malformed

	Notfound: INTEGER_32 = 12
			-- Unknown opcode in sqlite3_file_control()

	Full: INTEGER_32 = 13
			-- Insertion failed because database is full

	Cantopen: INTEGER_32 = 14
			-- Unable to open the database file

	Protocol: INTEGER_32 = 15
			-- Database lock protocol error

	Empty: INTEGER_32 = 16
			-- Internal use only

	Schema: INTEGER_32 = 17
			-- The database schema changed

	Toobig: INTEGER_32 = 18
			-- String or BLOB exceeds size limit

	Constraint: INTEGER_32 = 19
			-- Abort due to constraint violation

	Mismatch: INTEGER_32 = 20
			-- Data type mismatch

	Misuse: INTEGER_32 = 21
			-- Library used incorrectly

	Nolfs: INTEGER_32 = 22
			-- Uses OS features not supported on host

	Auth: INTEGER_32 = 23
			-- Authorization denied

	Format: INTEGER_32 = 24
			-- Not used

	Range: INTEGER_32 = 25
			-- 2nd parameter to sqlite3_bind out of range

	Row: INTEGER_32 = 100
