note
	description: "[
		Query result wrapper providing easy access to SQLite result rows.
		Automatically collects all rows for simple iteration.
	]"
	date: "$Date$"
	revision: "$Revision$"

class interface
	SIMPLE_SQL_RESULT

create 
	make,
	make_empty

feature -- Access

	rows: ARRAYED_LIST [SIMPLE_SQL_ROW]
			-- All result rows
	
feature -- Measurement

	count: INTEGER_32
			-- Number of rows in result
		ensure
			non_negative: Result >= 0
	
feature -- Status report

	is_empty: BOOLEAN
			-- Has no rows?
		ensure
			definition: Result = (count = 0)
	
feature -- Access

	first: SIMPLE_SQL_ROW
			-- First row
		require
			not_is_empty: not is_empty
		ensure
			result_attached: Result /= Void

	last: SIMPLE_SQL_ROW
			-- Last row
		require
			not_is_empty: not is_empty
		ensure
			result_attached: Result /= Void

	item alias "[]" (i: INTEGER_32): SIMPLE_SQL_ROW
			-- Row at index i
		require
			valid_index: i >= 1 and i <= count
		ensure
			result_attached: Result /= Void
	
invariant
	rows_attached: rows /= Void
