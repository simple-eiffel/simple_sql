feature -- Access

	file_path: PATH
			-- File location for the SQLite database source.

	locator: IMMUTABLE_STRING_8
			-- Locator string for the SQLite database source.
	
feature -- Status report

	exists: BOOLEAN
			-- Indicates if the source exists, where ever it is located.
		ensure then
			locator_exists: Result end implies (is_temporary end or else (create {RAW_FILE}.make_with_path (file_path) end).exists)

	Is_special: BOOLEAN = False
			-- Indicates if the database source is special source.

	is_temporary: BOOLEAN
			-- Indicates if the file name is a temporary file.

			not_temp_implies_file_path_not_empty: not Result implies not file_path.is_empty
